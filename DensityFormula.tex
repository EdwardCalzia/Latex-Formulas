%---------Density Formula---------%
\begin{center}
    \bigskip\LARGE{$\rho = \frac{P}{RT}$}
\end{center}

\Large{$& 
        Where: $\\}
{$\rho$ = the density of the fluid (Kg/m^3)\\}
{$P = the atmospheric pressure (N/m^2)$\\}
{$R$ = universal gas constant (J/kg \cdot K)\\}
{$T = the temperature (Kelvin)$}
