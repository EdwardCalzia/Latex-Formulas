%---------Reynolds Number---------%
\begin{center}
    \bigskip\LARGE{$Re = \frac{\rho u L}{\mu}$}
\end{center}

\Large{$& 
        Where: $\\}
{$Re$ = the Reynolds number (No Units)\\}
{$\rho$ = the density of the fluid (Kg/m^3)\\}
{$u = the flow speed (m/s)$\\}
{$L = the characteristic linear dimension (m)$\\}
{$\mu$ = the dynamic viscosity (N s/m2)\\}
