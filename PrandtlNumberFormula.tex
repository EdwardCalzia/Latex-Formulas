%---------Prandtl Number---------%
\begin{center}
    \bigskip\LARGE{$Pr = \frac{\mu / \rho}{k / (c_p \rho)}$}
\end{center}

\Large{$& 
        Where: $\\}
{$Pr$ = the Prandtl Number (No Units)\\}
{$\mu$ = dynamic viscosity (N s/m2)\\}
{$\rho$ = the density of the fluid (Kg/m^3)\\}
{$k = the thermal conductivity (W/(m·K))$\\}
{$c_p$ = Specific heat capacity (J/(kg·K))\\}
